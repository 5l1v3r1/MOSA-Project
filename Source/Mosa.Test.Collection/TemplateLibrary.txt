	Integers.Add(new KeyValuePair<string, string>("int", "I4"));
	Integers.Add(new KeyValuePair<string, string>("long", "I8"));

	Integers.Add(new KeyValuePair<string, string>("uint", "U4"));

	List<KeyValuePair<string, string>> Numbers = new List<KeyValuePair<string, string>>(Integers);
